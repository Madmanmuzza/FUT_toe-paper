\begin{abstract}
We promote the Einstein–Hilbert (EH) matching for the BD framework to a theorem with boundary control, then validate on FLRW, Newtonian, and linear-wave controls. Failure of any acceptance threshold falsifies this scope.
\end{abstract}

\paragraph{Run configuration.}
Seeds $=8$; sizes $N\in\{2^{10},2^{11},2^{12},2^{13}\}$; medians/IQRs reported. $\ellc\sim(\mathrm{Vol}/N)^{1/4}$.

\section{Quadratic form and curvature surrogate}
Let $B_{\rm BDG}$ be the discrete d’Alembertian \citep{AslanbeigiEtAl2014}. Define
\[
S_{\rm BD}[\phi]=\frac{1}{2}\sum_{x,y}\phi(x)\,B_{\rm BDG}(x,y)\,\phi(y)
+\alpha\,S_{\rm bdry}+\beta\,S_{\rm dens},
\]
with $S_{\rm bdry}$ a boundary counterterm and $S_{\rm dens}$ an interval-density surrogate.
Benincasa–Dowker’s 4D layer combination $S^{(4)}$ provides a scalar-curvature proxy \citep{BenincasaDowker2010}.

\begin{theorem}[EH matching with boundary control]\label{thm:EH}
On 4D Minkowski and weakly curved conformal controls $g=\Omega^2\eta$ with $\Omega\in C^3$ slowly varying, there exist constants $(G,\Lambda)$ and coefficients $(\alpha,\beta)$ such that, for compactly supported $\phi\in C_c^3$,
\[
S_{\rm BD}[\phi]=\int \Big( \tfrac{1}{2}\,\phi\,\Box_g \phi \Big)\sqrt{-g}\,d^4x
+ \int \Big(\tfrac{R}{16\pi G}-\Lambda\Big)\sqrt{-g}\,d^4x
+ O(\ellc^{\,p}),
\]
for some $p>0$, uniformly on finite regions after shaving a boundary collar of thickness $\Theta(\ellc)$.
\end{theorem}

\noindent\emph{Proof sketch.}
(i) By Paper II’s Theorem, $B_{\rm BDG}\to\Box$ in $L^2$ with rate $O(\ellc^{\,p})$. (ii) Replace layer sums by lightcone integrals and match coefficients to recover the continuum quadratic form; residuals are controlled by moment cancellations (Aslanbeigi–Saravani–Sorkin) and smoothness of $\phi$. (iii) Choose $(\alpha,\beta)$ to cancel boundary/discretization biases tied to “dangling” intervals near $\partial D$. (iv) Weak curvature: linearize in $\Omega-1$ to capture $R$ and $\Lambda$ terms; the BD density surrogate tracks $\sqrt{-g}$ to first order. \qed

\section{Controls and acceptance criteria}

\subsection{FLRW (Friedmann residuals)}
\textbf{Acceptance:} median absolute residual of the Hamiltonian constraint decreases with refinement; log–log slope $>0.5$ (95\% CI).
\begin{figure}[H]\centering
\SafeGraphic{friedmann_check.png}{0.80}
\caption{FLRW residuals vs $\ellc$. \textbf{Falsifier:} nonshrinking or nonmonotone residuals.}
\end{figure}

\subsection{Newtonian limit (Poisson residuals)}
\textbf{Acceptance:} $L^2$ error of the BD Laplacian surrogate vs Poisson $\propto \ellc^{\,p}$ with $p>0.5$ (95\% CI).
\begin{figure}[H]\centering
\SafeGraphic{poisson_residuals.png}{0.80}
\caption{Poisson residuals. \textbf{Falsifier:} exponent $\le 0$.}
\end{figure}

\subsection{Linear waves (dispersion)}
\textbf{Acceptance:} dispersion error $<5\%$ over a mid-$k$ window at top size; error decreases with $N$.
\begin{figure}[H]\centering
\SafeGraphic{dispersion_waves.png}{0.80}
\caption{Wave dispersion. \textbf{Falsifier:} $>5\%$ error or no refinement trend.}
\end{figure}
