\title{Paper VI: Gauge and Fermion Sectors\\
on Negentropic Birth--Death Causal Sets}
\author{Daniel J. Murray}
\date{\today}

\begin{document}
\maketitle

\begin{abstract}
We extend the BD causal-set framework to $U(1)$/$SU(N)$ gauge transport on Hasse links and to fermions with a discrete Dirac kernel aligned to the BDG lightcone. We define Wilson-loop and plaquette diagnostics, test dispersion, and perform anomaly-matching checks. Each diagnostic has an explicit acceptance threshold; failure of any threshold falsifies this sector at the stated scope.
\end{abstract}

\paragraph{Run configuration.}
Seeds $=8$; size ladder $N\in\{2^{10},2^{11},2^{12},2^{13}\}$ on flat causal diamonds; medians and IQRs reported. Coarse length $\ellc\sim(\mathrm{Vol}/N)^{1/4}$. Gauge groups tested: $U(1)$ (baseline), $SU(2)$ (optional). Link variables $U(x\!\to\!y)\in G$ assigned on cover relations.

\section{Gauge transport and curvature}
Assign $U(x\!\to\!y)$ on Hasse links and define the holonomy of a minimal closed loop $C$ in a small causal diamond by ordered multiplication. The Wilson loop is
\[
W(C)=\frac{1}{\mathrm{dim}}\mathrm{Re}\,\mathrm{tr}\Big(\prod_{e\in C}U(e)\Big).
\]
\noindent\textbf{Acceptance (Wilson scaling):} on smooth controls, $\log\langle W(C)\rangle$ vs area $A$ exhibits the expected regime (area law at strong coupling; perimeter-like at weak) with slope stable under refinement (slope change $<10\%$ across the top two sizes).

\begin{figure}[H]\centering
\SafeGraphic{wilson_loop_scaling.png}{0.80}
\caption{Wilson loop scaling. \textbf{Falsifier:} slope flips regime under refinement or varies $>10\%$ across the top sizes.}
\end{figure}

\subsection*{Plaquette spectrum}
For “plaquettes” (minimal causal-diamond loops), let $\theta$ be the gauge-invariant angle (Abelian) or an $SU(N)$ class angle. \textbf{Acceptance (smoothing):} the distribution narrows with refinement on smooth controls; interquartile width decreases monotonically with $N$.

\begin{figure}[H]\centering
\SafeGraphic{plaquette_hist.png}{0.80}
\caption{Plaquette angle distribution. \textbf{Falsifier:} no narrowing trend (IQR non-decreasing) under refinement.}
\end{figure}

\section{Fermions and the Dirac kernel}
Define a discrete Dirac operator $D$ acting on spinors $\psi:\mathcal{C}\to\mathbb{C}^4$ using directed incidence maps aligned to BDG lightcone structure; mass term $m\bar\psi\psi$ and gauge-covariant transport via link $U$ enter in the standard way. Compare discrete energies $E(k)$ with $E_{\mathrm{cont}}(k)=\sqrt{k^2+m^2}$ over mid-momentum windows.

\noindent\textbf{Acceptance (dispersion):} median relative error $<5\%$ on the mid-$k$ window at top size; error decreases with refinement (significant downward slope at 95\% CI).

\begin{figure}[H]\centering
\SafeGraphic{fermion_dispersion.png}{0.80}
\caption{Fermion dispersion. \textbf{Falsifier:} $>5\%$ error at top size or non-decreasing error with $N$.}
\end{figure}

\section{Chiral issues and anomaly matching}
Nielsen–Ninomiya obstructions \citep{NielsenNinomiya1981} constrain chiral regularizations. We test a parity-preserving doubled construction with counterterms and measure the divergence of the chiral current in background $U(1)$ fields.

\noindent\textbf{Acceptance (anomaly):} anomaly ratio (measured/expected) $\to 1$ with refinement; linear fit vs resolution has positive slope with 95\% CI excluding $0$.

\begin{figure}[H]\centering
\SafeGraphic{anomaly_match.png}{0.80}
\caption{Chiral anomaly matching. \textbf{Falsifier:} ratio fails to approach $1$; slope CI includes $0$.}
\end{figure}

\section{Small-diamond loops and gauge invariance}
We characterize minimal loops in Hasse diagrams and prove $W(C)$ invariance under local $G$ transformations at vertices.

\section{Dirac construction on a partial order}
We sketch $D$ from directed incidence, with locality and symmetry constraints adapted from BDG kernels \citep{AslanbeigiEtAl2014}. Coupling to curvature surrogates leverages interval-abundance weights \citep{BenincasaDowker2010}.
