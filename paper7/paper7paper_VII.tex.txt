\begin{abstract}
We confront the BD causal-set framework with precision cosmology. A minimal BD$\to$FLRW map defines effective parameters $(H_0,\Omega_m,\Omega_\Lambda,n_s,\sigma_8)$. We quantify CMB, LSS, BBN, and SGWB signatures and impose explicit acceptance thresholds (falsifiers). Any threshold failure rules out the model at this scope; all deviations must shrink with coarse length $\ellc\sim(\mathrm{Vol}/N)^{1/4}$.
\end{abstract}

\paragraph{Run configuration.}
Seeds $=8$; size ladder $N\in\{2^{10},2^{11},2^{12},2^{13}\}$. For each $N$, construct Minkowski/FLRW controls and compute medians and IQRs across seeds. All reported deviations are functions of $\ellc$ with bootstrap 95\% CIs.

We analyze TT residuals $\Delta C_\ell/C_\ell$ against Planck~2018 \citep{Planck2018}.

\section{Large-scale structure (galaxy clustering)}
We compute the matter power ratio $R(k)=P_{\mathrm{BD}}(k)/P_{\Lambda\mathrm{CDM}}(k)$ and compare to BOSS DR12 scales \citep{BOSSDR12}.
\begin{figure}[H]\centering
\SafeGraphic{matter_power_ratio.png}{0.80}
\caption{Matter power ratio $R(k)$. \textbf{Acceptance:} broad deviations $\lesssim 2\%$ on $k\in[0.02,0.2]\ h\,\mathrm{Mpc}^{-1}$ that move toward $R\!=\!1$ as $\ellc\to0$. \textbf{Falsifier:} a fixed, nonshrinking feature at physical $k$.}
\end{figure}

\section{Big Bang Nucleosynthesis (BBN)}

\section{Stochastic gravitational-wave background (SGWB)}
