\title{Paper V: Renormalization and Continuum EFT\\
on Negentropic Birth--Death Causal Sets}
\author{Daniel J. Murray}

\begin{abstract}
We define a label-invariant blocking map for fields on BD causal sets, establish an approximate semigroup property with controlled error, and show stability of running couplings under reblocking. We verify spectral positivity and demonstrate finite-size scaling collapse. Each diagnostic has explicit acceptance thresholds; failure falsifies this scope.
\end{abstract}

\paragraph{Run configuration.}
Seeds $=8$; $N\in\{2^{10},2^{11},2^{12},2^{13}\}$; blocks $n_b\in\{16,32,64\}$; medians/IQRs reported. $\ellc\sim(\mathrm{Vol}/N)^{1/4}$.

\section{Blocking map and couplings}
Partition into order-interval blocks of mean size $n_b$; define coarse fields by block averages and infer couplings $\mathbf{g}=(m^2,\lambda,\ldots)$ from two-/four-point summaries.

\begin{proposition}[Approximate semigroup]\label{prop:semi}
Let $\mathcal{B}_{n_b}$ be the interval-block map. There exists $\epsilon(n_b)=O(n_b^{-1/2})$ such that
\[
\mathcal{B}_{n_b}\circ \mathcal{B}_{m_b} = \mathcal{B}_{n_b m_b}\circ (I + \Delta_{n_b,m_b}),
\qquad \|\Delta_{n_b,m_b}\| \le \epsilon(n_b)+\epsilon(m_b),
\]
in operator norm on correlation summaries, uniformly over seeds on finite diamonds.
\end{proposition}

\noindent\emph{Proof sketch.}
Concentration of block statistics around interval means yields $O(n_b^{-1/2})$ fluctuations; composition errors add subadditively. \qed

\begin{theorem}[Reblocking stability of running couplings]\label{thm:stab}
Fix $\mu\equiv \ellc^{-1}$. Under Proposition~\ref{prop:semi}, the coefficient of variation of $g_i(\mu)$ over $n_b\in\{16,32,64\}$ obeys
\[
\mathrm{CV}[g_i(\mu)] \le C\,(\epsilon(16)+\epsilon(32)+\epsilon(64)) = O(n_b^{-1/2}),
\]
hence $\mathrm{CV}\to 0$ as refinement proceeds and blocks grow.
\end{theorem}

\section{Running couplings and fixed points}
Estimate $g(\mu)$ across the ladder; compute $\beta_i=\ellc\,\Delta g_i/\Delta\ellc$ and locate fixed points; classify eigenvalues of $M_{ij}=\partial\beta_i/\partial g_j$.
