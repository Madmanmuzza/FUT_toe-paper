\title{Paper VIII: Black Holes and Semiclassical Thermodynamics\\
on Negentropic Birth--Death Causal Sets}
\author{Daniel J. Murray}
\date{\today}

\begin{document}
\maketitle

\begin{abstract}
We develop black-hole thermodynamics and semiclassical radiation for the BD causal-set framework. We extract Hawking temperature $T=\kappa/2\pi$ from near-horizon response, confirm the Bekenstein--Hawking area law $S=A/4$, compute quasinormal spectra from BD operators, reconstruct Page curves for unitary evaporation, and estimate greybody factors. Each diagnostic has an explicit acceptance threshold; any failure falsifies this sector at the stated scope.
\end{abstract}

\paragraph{Run configuration.}
Seeds $=8$; size ladder $N\in\{2^{10},2^{11},2^{12},2^{13}\}$. Near-horizon (Rindler) controls for temperature, stationary effective background for QNMs, and evaporating toy background for Page curves. Medians and IQRs reported across seeds.

\section{Hawking temperature from near-horizon response}
A BD Unruh--DeWitt detector at proper acceleration $a$ measures a Planckian spectrum with $T=a/2\pi$; mapping to a Killing horizon with surface gravity $\kappa$ yields $T=\kappa/2\pi$.

\noindent\textbf{Acceptance (slope):} fit $T$ vs $\kappa$; relative slope error $<5\%$ at top size; error decreases with refinement (95\% CI excludes non-decreasing trend).

\begin{figure}[H]\centering
\SafeGraphic{hawking_temperature.png}{0.80}
\caption{Hawking temperature vs surface gravity. \textbf{Falsifier:} slope differs from $1/2\pi$ by $\ge 5\%$ and does not shrink with $N$.}
\end{figure}

\section{Area law $S=A/4$}
We estimate microstate entropy from BD growth with boundary contributions and fit $S(A)$.

\noindent\textbf{Acceptance (area slope):} linear fit slope $=1/4\pm 5\%$ at top size; residuals decrease with refinement.

\begin{figure}[H]\centering
\SafeGraphic{area_law_entropy.png}{0.80}
\caption{Entropy vs horizon area. \textbf{Falsifier:} slope deviates from $1/4$ beyond 5\% or residuals fail to shrink with $N$.}
\end{figure}

\section{Ringdown: quasinormal modes}
Perturbations of the BD effective operator define complex frequencies $\omega_n$ characterizing ringdown.

\noindent\textbf{Acceptance (spectral trend):} real-part spacing and damping trend consistent with continuum expectations \citep{BertiCardosoStarinets2009} within 10\%, improving with refinement.

\begin{figure}[H]\centering
\SafeGraphic{quasinormal_spectrum.png}{0.80}
\caption{Quasinormal spectrum: ${\rm Re}\,\omega$ vs $-{\rm Im}\,\omega$. \textbf{Falsifier:} spacings/damping inconsistent and not improving with $N$.}
\end{figure}

\section{Information flow: Page curve}
Using BD correlators, we partition radiation vs interior and estimate entanglement entropy $S_{\rm rad}(t)$.

\noindent\textbf{Acceptance (unitary turnover):} clear turnover near $t\simeq t_{\rm evap}/2$ with strong-subadditivityâ€“consistent shape \citep{Page1993}; turnover sharpens with refinement.

\section{Greybody factors}
We compute barrier transmission coefficients $\Gamma_\ell(\omega)$ from effective scattering of BD perturbations.

\noindent\textbf{Acceptance (asymptotics):} low-$\omega$ suppression and high-$\omega$ saturation match continuum scaling; deviations shrink with $N$.

\end{figure}
